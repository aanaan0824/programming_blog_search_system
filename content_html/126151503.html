"""
对如下DF，设置两个单元格的值
·使用iloc 设置(3,B)的值是nan
·使用loc设置(8,D)的值是nan
"""
import numpy as np
import pandas as pd

np.random.seed(66)
df = pd.DataFrame(np.random.rand(10, 4), columns=list('ABCD'))
df.iloc[3, 1] = np.nan
df.loc[8, 'D'] = np.nan
print(df)


print(df.isnull().sum())
