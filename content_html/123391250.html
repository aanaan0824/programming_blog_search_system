//逆置函数 
void reverse(int* nums, int left, int right) 
{
    while(left < right)
    {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
        left++;
        right--;
    } 
}

void rotate(int* nums, int numsSize, int k) 
{
    k %= numsSize;//保证K的合法性
    //第一步：后K个逆置
    reverse(nums, numsSize - k, numsSize - 1);
    //第二步：前N-K个逆置
    reverse(nums, 0, numsSize - k - 1);
    //第三步：整体逆置
    reverse(nums, 0, numsSize - 1); 
}
