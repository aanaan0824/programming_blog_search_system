对象传参,注意参数值是个对象

let param=={

id:1

}

axios({

url :"xxxxxxx"，

method:"post"

data:param,

})

字符串传参，要加一个 headers：{ "content-type" : "text/plain" } ，注意参数值是个字符串

let param=={

id:1

}

axios({

url :"xxxxxxx"，

method:"post"

headers：{ "content-type" : "text/plain" }

data:param.id,

})

键值对传参，将data改为params

let param=={

id:1

}

axios({

url :"xxxxxxx"，

method:"post"

params:param,

})
